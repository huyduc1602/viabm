CREATE TABLE category_blog(
	id int PRIMARY KEY AUTO_INCREMENT NOT NULL,
	name varchar(255) NOT NULL,
    slug varchar(255) NULL,
	parentId int NOT NULL default(0)
);

CREATE TABLE blog (
	id int PRIMARY KEY AUTO_INCREMENT NOT NULL,
	name varchar(255) NOT NULL,
	slug varchar(255) NULL,
	image varchar(255) NULL,
	detail text NULL,
	sumary varchar(255) NULL,
	categoryId int NOT NULL,
	userId int NOT NULL,
	display varchar(4) DEFAULT('HIDE'),
	createdDate datetime NOT NULL default(now()),
	updatedDate datetime NOT NULL default(now()),
    FOREIGN KEY (categoryId) REFERENCES category_blog(id)
);

CREATE TABLE blog (
	id int PRIMARY KEY AUTO_INCREMENT NOT NULL,
	name varchar(255) NOT NULL,
	name_en varchar(255) NOT NULL,
	slug varchar(255) NULL,
	image varchar(255) NULL,
	detail text NULL,
    detail_en text NULL,
	sumary varchar(255) NULL,
    sumary_en varchar(255) NULL,
	categoryId int NOT NULL,
	userId int NOT NULL,
	display varchar(4) DEFAULT('HIDE'),
    view_count int DEFAULT(0),
    last_read datetime NOT NULL default(now()),
	createdDate datetime NOT NULL default(now()),
	updatedDate datetime NOT NULL default(now()),
    FOREIGN KEY (categoryId) REFERENCES category_blog(id)
);
